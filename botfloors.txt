        let selectedMaterial = null;

        function renderGrid() {
            const grid = document.getElementById('grid');
            grid.innerHTML = '';
            materials.forEach(m => {
                const card = document.createElement('div');
                card.className = 'card';
                card.innerHTML = `
                    <span class="emoji">${m.emoji}</span>
                    <div class="name">${m.name}</div>
                    <span class="badge">${m.badge}</span>
                `;
                card.onclick = () => showDetail(m.id);
                grid.appendChild(card);
            });
        }

        function showDetail(id) {
            selectedMaterial = id;
            const data = details[id];
            const material = materials.find(m => m.id === id);

            document.getElementById('grid').classList.add('hidden');
            document.getElementById('detail').classList.remove('hidden');

            document.getElementById('detailName').textContent = `${material.emoji} ${material.name}`;
            document.getElementById('detailDesc').textContent = data.desc;
            document.getElementById('detailPros').textContent = data.pros;
            document.getElementById('detailCons').textContent = data.cons;
            document.getElementById('detailPrice').textContent = data.price;

            // Настройка кнопок
            document.getElementById('askBtn').onclick = () => {
                tg.sendData(JSON.stringify({ action: 'ask', material: id }));
                tg.close();
            };
            document.getElementById('compareBtn').onclick = () => {
                tg.sendData(JSON.stringify({ action: 'compare', material: id }));
                tg.close();
            };

            tg.HapticFeedback.impactOccurred('medium');
        }

        document.getElementById('backBtn').onclick = () => {
            document.getElementById('grid').classList.remove('hidden');
            document.getElementById('detail').classList.add('hidden');
            selectedMaterial = null;
        };

        // Обработка данных от бота (если нужно)
        tg.onEvent('mainButtonClicked', () => {
            if (selectedMaterial) {
                tg.sendData(JSON.stringify({ action: 'select', material: selectedMaterial }));
                tg.close();
            }
        });

        // Инициализация
        renderGrid();
